// Sender Email: sender@example.com
// Recipient Email: recipient@example.com
// Message Body: Hello, this is a secure S/MIME message. Please verify the signature and decrypt the content.

import java.security.KeyPair;
import java.security.KeyPairGenerator;
import java.security.PrivateKey;
import java.security.PublicKey;
import java.security.cert.X509Certificate;
import java.util.Properties;
import javax.mail.*;
import javax.mail.internet.*;

public class SMIMEMessageExample {

    public static MimeMessage createSMIMEMessage() throws Exception {
        // Set up properties (customize as needed)
        Properties properties = System.getProperties();
        Session session = Session.getDefaultInstance(properties);

        // Generate a self-signed certificate (for demonstration purposes)
        KeyPairGenerator keyPairGenerator = KeyPairGenerator.getInstance("RSA");
        keyPairGenerator.initialize(2048);
        KeyPair keyPair = keyPairGenerator.generateKeyPair();
        X509Certificate certificate = CertificateUtils.generateSelfSignedCertificate(keyPair);

        // Create a new MimeMessage
        MimeMessage message = new MimeMessage(session);
        message.setFrom(new InternetAddress("sender@example.com"));
        message.addRecipient(Message.RecipientType.TO, new InternetAddress("recipient@example.com"));
        message.setSubject("Secure S/MIME Message");

        // Construct the message body (plaintext or HTML)
        String messageBody = "Hello, this is a secure S/MIME message. Please verify the signature and decrypt the content.";

        // Sign the message (attach the signature)
        SMIMESignedGenerator signedGenerator = new SMIMESignedGenerator();
        signedGenerator.addSigner(keyPair.getPrivate(), certificate, SMIMESignedGenerator.DIGEST_SHA256);
        MimeMultipart signedContent = signedGenerator.generate(messageBody.getBytes());

        // Encrypt the message (attach the encrypted content)
        X509Certificate recipientCertificate = CertificateUtils.loadRecipientCertificate("recipient.cer"); // Load recipient's certificate
        SMIMEEnvelopedGenerator envelopedGenerator = new SMIMEEnvelopedGenerator();
        envelopedGenerator.addRecipientInfoGenerator(new JceKeyTransRecipientInfoGenerator(recipientCertificate));
        MimeBodyPart encryptedContent = envelopedGenerator.generate(signedContent);

        // Set the overall content (signed and encrypted)
        message.setContent(encryptedContent);

        return message;
    }

    public static void main(String[] args) {
        try {
            MimeMessage smimeMessage = createSMIMEMessage();
            // Send the smimeMessage via SSL/TLS socket
            // ...
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
